cmake_minimum_required(VERSION 3.16)
set(IDF_TARGET esp32p4)

set(AUDIOSENSE_DEFAULT_IDF_PATH "E:/ESPIDF5.4.4/.espressif/v5.4.4/esp-idf")
if(DEFINED ENV{IDF_PATH} AND EXISTS "$ENV{IDF_PATH}/tools/cmake/project.cmake")
    set(AUDIOSENSE_IDF_PATH "$ENV{IDF_PATH}")
elseif(EXISTS "${AUDIOSENSE_DEFAULT_IDF_PATH}/tools/cmake/project.cmake")
    set(ENV{IDF_PATH} "${AUDIOSENSE_DEFAULT_IDF_PATH}")
    set(AUDIOSENSE_IDF_PATH "${AUDIOSENSE_DEFAULT_IDF_PATH}")
else()
    message(FATAL_ERROR "未找到有效 ESP-IDF。当前 IDF_PATH='$ENV{IDF_PATH}'，默认路径='${AUDIOSENSE_DEFAULT_IDF_PATH}'。请先配置 ESP-IDF 插件或在 ESP-IDF 终端中运行。")
endif()
include("${AUDIOSENSE_IDF_PATH}/tools/cmake/project.cmake")
project(audiosense-glasses)

set(AUDIOSENSE_ESPDL_MODEL "${CMAKE_CURRENT_SOURCE_DIR}/models/audiosense.espdl")
if(EXISTS "${AUDIOSENSE_ESPDL_MODEL}")
    esptool_py_flash_to_partition(flash espdl_model "${AUDIOSENSE_ESPDL_MODEL}")
else()
    message(FATAL_ERROR "Missing ESP-DL model: ${AUDIOSENSE_ESPDL_MODEL}")
endif()
